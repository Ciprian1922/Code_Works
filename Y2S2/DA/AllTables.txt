CREATE TABLE COURSE (
    COURSE_NO VARCHAR2(8 CHAR) NOT NULL,
    DESCRIPTION VARCHAR2(50 CHAR),
    COST NUMBER(8, 2),
    PREREQUISITE VARCHAR2(8 CHAR),
    CREATED_BY VARCHAR2(30 CHAR) NOT NULL,
    CREATED_DATE DATE NOT NULL,
    MODIFIED_BY VARCHAR2(30 CHAR) NOT NULL,
    MODIFIED_DATE DATE NOT NULL,
    
    CONSTRAINT PK_COURSE PRIMARY KEY (COURSE_NO)
);

CREATE TABLE INSTRUCTOR (
    INSTRUCTOR_ID NUMBER(8) NOT NULL,
    SALUTATION_ID VARCHAR2(5 CHAR) NOT NULL,
    FIRST_NAME VARCHAR2(25 CHAR) NOT NULL,
    LAST_NAME VARCHAR2(25 CHAR) NOT NULL,
    STREET_ADDRESS VARCHAR2(50 CHAR) NOT NULL,
    ZIP VARCHAR2(15 CHAR) NOT NULL,
    PHONE VARCHAR2(15 CHAR) NOT NULL,
    CREATED_BY VARCHAR2(30 CHAR) NOT NULL,
    CREATED_DATE DATE NOT NULL,
    MODIFIED_BY VARCHAR2(30 CHAR) NOT NULL,
    MODIFIED_DATE DATE NOT NULL,
    
    CONSTRAINT PK_INSTRUCTOR PRIMARY KEY (INSTRUCTOR_ID)
);

CREATE TABLE ZIPCODE (
    ZIP VARCHAR2(5 CHAR) NOT NULL,
    CITY VARCHAR2(25 CHAR),
    STATE VARCHAR2(2 CHAR),
    CREATED_BY VARCHAR2(30 CHAR) NOT NULL,
    CREATED_DATE DATE NOT NULL,
    MODIFIED_BY VARCHAR2(30 CHAR) NOT NULL,
    MODIFIED_DATE DATE NOT NULL,
    
    CONSTRAINT PK_ZIPCODE PRIMARY KEY (ZIP)
);

CREATE TABLE SECTION (
    SECTION_ID NUMBER(8,0) NOT NULL,
    COURSE_NO VARCHAR2(8 CHAR) NOT NULL,
    SECTION_NO NUMBER(3,0) NOT NULL,
    START_DATE_TIME DATE NOT NULL,
    LOCATION VARCHAR2(50 CHAR),
    INSTRUCTOR_ID NUMBER (8,0) NOT NULL,
    CAPACITY NUMBER (3,0),
    CREATED_BY VARCHAR2 (30 CHAR) NOT NULL,
    CREATED_DATE DATE NOT NULL,
    MODIFIED_BY VARCHAR2 (30 CHAR) NOT NULL,
    MODIFIED_DATE DATE NOT NULL,

    CONSTRAINT PK_SECTION PRIMARY KEY (SECTION_ID),
    CONSTRAINT FK_SECTION_COURSE FOREIGN KEY (COURSE_NO)
        REFERENCES COURSE(COURSE_NO)
);

CREATE TABLE STUDENT (
    STUDENT_ID NUMBER(8, 0) NOT NULL,
    SALUTATION VARCHAR2(25 CHAR),
    FIRST_NAME VARCHAR2(25 CHAR) NOT NULL,
    LAST_NAME VARCHAR2(25 CHAR) NOT NULL,
    STREET_ADDRESS VARCHAR2(50 CHAR) NOT NULL,
    ZIP VARCHAR2(15 CHAR) NOT NULL,
    PHONE VARCHAR2(15 CHAR) NOT NULL,
    EMPLOYER VARCHAR2(50 CHAR),
    REGISTRATION_DATE DATE NOT NULL,
    CREATED_BY VARCHAR2(30 CHAR) NOT NULL,
    CREATED_DATE DATE NOT NULL,
    MODIFIED_BY VARCHAR2(30 CHAR) NOT NULL,
    MODIFIED_DATE DATE NOT NULL,
    
    CONSTRAINT PK_STUDENT PRIMARY KEY (STUDENT_ID)
);

CREATE TABLE ENROLLMENT (
    STUDENT_ID NUMBER(8, 0) NOT NULL,
    SECTION_ID NUMBER(8, 0) NOT NULL,
    ENROLL_DATE DATE NOT NULL,
    FINAL_GRADE VARCHAR2(2 CHAR),
    CREATED_BY VARCHAR2(30 CHAR) NOT NULL,
    CREATED_DATE DATE NOT NULL,
    MODIFIED_BY VARCHAR2(30 CHAR) NOT NULL,
    MODIFIED_DATE DATE NOT NULL,

    CONSTRAINT PK_ENROLLMENT PRIMARY KEY (STUDENT_ID, SECTION_ID),
    CONSTRAINT FK_ENROLLMENT_STUDENT FOREIGN KEY (STUDENT_ID)
        REFERENCES STUDENT(STUDENT_ID),
    CONSTRAINT FK_ENROLLMENT_SECTION FOREIGN KEY (SECTION_ID)
        REFERENCES SECTION(SECTION_ID)
);

CREATE TABLE GRADE_TYPE (
    GRADE_TYPE_CODE CHAR(2) NOT NULL,
    DESCRIPTION VARCHAR2(50 CHAR) NOT NULL,
    CREATED_BY VARCHAR2(30 CHAR) NOT NULL,
    CREATED_DATE DATE NOT NULL,
    MODIFIED_BY VARCHAR2(30 CHAR) NOT NULL,
    MODIFIED_DATE DATE NOT NULL,
    
    CONSTRAINT PK_GRADE_TYPE PRIMARY KEY (GRADE_TYPE_CODE)
);

CREATE TABLE GRADE_TYPE_WEIGHT (
    SECTION_ID NUMBER(8) NOT NULL,
    GRADE_TYPE_CODE CHAR(2) NOT NULL,
    NUMBER_PER_SECTION NUMBER(3) NOT NULL,
    PERCENT_OF_FINAL_GRADE NUMBER(5,2) NOT NULL,
    DROP_LOWEST CHAR(1),
    CREATED_BY VARCHAR2(30 CHAR) NOT NULL,
    CREATED_DATE DATE NOT NULL,
    MODIFIED_BY VARCHAR2(30 CHAR) NOT NULL,
    MODIFIED_DATE DATE NOT NULL,

    CONSTRAINT PK_GRADE_TYPE_WEIGHT PRIMARY KEY (SECTION_ID, GRADE_TYPE_CODE),
    CONSTRAINT FK_GRADE_TYPE_WEIGHT_SECTION FOREIGN KEY (SECTION_ID)
        REFERENCES SECTION(SECTION_ID),
    CONSTRAINT FK_GRADE_TYPE_WEIGHT_GRADE_TYPE FOREIGN KEY (GRADE_TYPE_CODE)
        REFERENCES GRADE_TYPE(GRADE_TYPE_CODE)
);
